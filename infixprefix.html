Infix to Prefix

Infix : A + B * C

Ilustrasi

Implementasi Python


def infix_to_prefix(expression):
    expression = expression[::-1]

    stack = []
    output = []
    precedence = {'+':1, '-':1, '*':2, '/':2}

    for char in expression:
        if char.isalnum():
            output.append(char)
        elif char == ')':
            stack.append(char)
        elif char == '(':
            while stack[-1] != ')':
                output.append(stack.pop())
            stack.pop()
        else:
            while (stack and stack[-1] != ')' and 
                   precedence[char] < precedence[stack[-1]]):
                output.append(stack.pop())
            stack.append(char)

    while stack:
        output.append(stack.pop())

    return "".join(output[::-1])

print(infix_to_prefix("A+B*C"))
